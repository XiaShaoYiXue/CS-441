  filter(gender == "Female", 
         !is.na(medium),
         medium != "",
         !str_trim(medium) == "") %>%
  group_by(medium) %>%
  summarise(count = n()) %>%
  top_n(10, count) %>%
  ggplot(aes(x = reorder(medium, count), y = count)) +
  geom_bar(stat = "identity", fill = "skyblue") +
  coord_flip() +
  theme_minimal() +
  labs(title = "Top 10 Mediums Used by Male Artists (Tate)",
       x = "Medium",
       y = "Number of Artworks")
